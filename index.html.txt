<script>
    const quizQuestions = [
        {
            type: "boolean",
            question: "Слово «Мышечная ткань» переводится на узбекский язык как «Mushak to'qimasi».",
            options: ["Верно", "Неверно"],
            correctAnswer: "Верно"
        },
        {
            type: "boolean",
            question: "Слово «Нервная система» переводится на узбекский язык как «Asab hujayrasi».",
            options: ["Верно", "Неверно"],
            correctAnswer: "Неверно"
        },
        {
            type: "boolean",
            question: "Выражение «Побочные действие» переводится как «Muolaja ta'siri».",
            options: ["Верно", "Неверно"],
            correctAnswer: "Неверно"
        },
        {
            type: "select",
            question: "Какая медицинская специализация занимается изучением и лечением заболеваний почек?",
            options: ["Вертебролог", "Nefrolog", "Трихолог", "Гематолог"],
            correctAnswer: "Nefrolog"
        },
        {
            type: "select",
            question: "Кто является первооткрывателем пенициллина?",
            options: ["Эдвард Дженнер", "Луи Пастер", "Александр Флеминг", "Роберт Кох"],
            correctAnswer: "Александр Флеминг"
        },
        {
            type: "select",
            question: "Какой врач специализируется на лечении заболеваний центральной и периферической нервной системы?",
            options: ["Гепатолог", "Nevrolog", "Иммунолог", "Пульмонолог"],
            correctAnswer: "Nevrolog"
        }
    ];

    function initQuiz() {
        const container = document.getElementById('quizContainer');
        if (!container) return;
        container.innerHTML = ''; 

        quizQuestions.forEach((q, qIndex) => {
            const card = document.createElement('div');
            card.className = 'question-card';
            card.id = 'question_' + qIndex;

            const numDiv = document.createElement('div');
            numDiv.className = 'question-number';
            numDiv.textContent = 'Вопрос ' + (qIndex + 1);
            card.appendChild(numDiv);

            const textDiv = document.createElement('div');
            textDiv.className = 'question-text';
            textDiv.textContent = q.question;
            card.appendChild(textDiv);

            const optionsList = document.createElement('ul');
            optionsList.className = 'options-list';

            q.options.forEach((option) => {
                const li = document.createElement('li');
                const label = document.createElement('label');
                label.className = 'option-label';

                const input = document.createElement('input');
                input.type = 'radio';
                input.name = 'question_input_' + qIndex;
                input.value = option;
                
                input.onchange = function() {
                    card.classList.remove('unanswered');
                };

                const span = document.createElement('span');
                span.textContent = option;

                label.appendChild(input);
                label.appendChild(span);
                li.appendChild(label);
                optionsList.appendChild(li);
            });

            card.appendChild(optionsList);
            container.appendChild(card);
        });
    }

    function checkQuiz() {
        const alertBanner = document.getElementById('alertBanner');
        const resultBox = document.getElementById('resultBox');
        if (alertBanner) alertBanner.style.display = 'none';
        
        let correctCount = 0;
        let hasUnanswered = false;

        quizQuestions.forEach((q, qIndex) => {
            const selectedInput = document.querySelector('input[name="question_input_' + qIndex + '"]:checked');
            const cardElement = document.getElementById('question_' + qIndex);

            if (!selectedInput) {
                if (cardElement) cardElement.classList.add('unanswered');
                hasUnanswered = true;
            } else {
                if (cardElement) cardElement.classList.remove('unanswered');
                if (selectedInput.value === q.correctAnswer) {
                    correctCount++;
                }
            }
        });

        if (hasUnanswered) {
            if (alertBanner) alertBanner.style.display = 'block';
            window.scrollTo({ top: 0, behavior: 'smooth' });
            return;
        }

        const totalCount = quizQuestions.length;
        const percentage = Math.round((correctCount / totalCount) * 100);

        document.getElementById('correctCount').textContent = correctCount;
        document.getElementById('totalCount').textContent = totalCount;
        document.getElementById('percentageValue').textContent = percentage + '%';

        if (resultBox) {
            resultBox.style.display = 'block';
            resultBox.scrollIntoView({ behavior: 'smooth', block: 'end' });
        }
    }

    document.addEventListener('DOMContentLoaded', initQuiz);
</script>
